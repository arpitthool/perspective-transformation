#version 460 core

layout (location = 0) in vec2 vertexPos;
layout (location = 1) in vec3 vertexColor;

out vec3 fragmentColor;

void main() {
    gl_Position = vec4(vertexPos, 0.0, 1.0);  // Set the position of the vertex in clip space , 
    // here z = 0.0 because we are in 2D, and w = 1.0 because we are using a perspective projection
    
    fragmentColor = vertexColor;
}
